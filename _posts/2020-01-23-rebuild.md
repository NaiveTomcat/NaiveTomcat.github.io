#include <iostream>
#include <fstream>
#include <string>

void mkfile(const char* timent)
{
fout.open(timent);

}

void head(string tit)
{
fout << R"(
---
layout: post
title:  )";
fout << tit << endl << "---\n\n";
return;

}

void ending()
{
fout << R"(
***

}

int main()
{

time_t now = time(0);
char* dt = ctime(&now);


cout << "Naive Tomcat Log Entry System\n" << dt << endl;

string title;

cout << "Please enter your file name, like yyyy-mm-dd-title.md\n";
cin >> Filename;
mkfile(Filename);
cout << "Enter your title\n";
cin >> title;
head(title);

while (temp != "EOT EOT EOT EOT EOT")

fout << temp << endl;

}
